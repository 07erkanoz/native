import android.content.Context;

import com.facebook.react.modules.core.DeviceEventManagerModule;

public class ProximitySensorModule extends ReactContextBaseJavaModule implements SensorEventListener {

    private SensorManager sensorManager;
    private Sensor proximitySensor;
    private PowerManager powerManager;
    private PowerManager.WakeLock wakeLock;
    private ReactApplicationContext reactContext;
    private boolean isEnabled = false;

    // Hassasiyet eÅŸik deÄŸerleri
    private float thresholdLow = 1.0f;
    private float thresholdNormal = 3.0f;
    private float thresholdHigh = 5.0f;
    private float currentThreshold = thresholdNormal;

    public ProximitySensorModule(ReactApplicationContext context) {
        super(context);
        this.reactContext = context;

        sensorManager = (SensorManager) context.getSystemService(Context.SENSOR_SERVICE);
        proximitySensor = sensorManager.getDefaultSensor(Sensor.TYPE_PROXIMITY);
        powerManager = (PowerManager) context.getSystemService(Context.POWER_SERVICE);

        // PROXIMITY_SCREEN_OFF_WAKE_LOCK - ekranÄ± kapatmak iÃ§in
        if (android.os.Build.VERSION.SDK_INT >= 21) {

            );
        }

    public String getName() {
        return "ProximitySensorModule";
    }

    @ReactMethod
    public void enable() {
        if (proximitySensor != null && !isEnabled) {
            sensorManager.registerListener(
                this,
                proximitySensor,
                SensorManager.SENSOR_DELAY_NORMAL
            );
            isEnabled = true;
        }
    }

    @ReactMethod
    public void disable() {
        if (isEnabled) {
            sensorManager.unregisterListener(this);
            if (wakeLock != null && wakeLock.isHeld()) {
                wakeLock.release();
            }
            isEnabled = false;
        }
    }

    @ReactMethod
    public void setSensitivity(String sensitivity) {
        switch (sensitivity) {
            case "low":
                currentThreshold = thresholdLow;
                break;
            case "high":
                currentThreshold = thresholdHigh;
                break;
            default:
                currentThreshold = thresholdNormal;
        }
    }

    @ReactMethod
    public void turnOffScreen() {
        if (wakeLock != null && !wakeLock.isHeld()) {
            wakeLock.acquire();
        }
    }

    @ReactMethod
    public void turnOnScreen() {
        if (wakeLock != null && wakeLock.isHeld()) {
            wakeLock.release();
        }
    }

    public void onSensorChanged(SensorEvent event) {
        if (event.sensor.getType() == Sensor.TYPE_PROXIMITY) {
            float distance = event.values[0];

            params.putDouble("timestamp", System.currentTimeMillis());

            // EkranÄ± kontrol et
            if (isNear) {
                turnOffScreen();
            } else {
                turnOnScreen();
            }
        }
    }

        // Accuracy deÄŸiÅŸikliÄŸi
    }

    private void sendEvent(String eventName, WritableMap params) {
        reactContext
            .getJSModule(DeviceEventManagerModule.RCTDeviceEventEmitter.class)
            .emit(eventName, params);
    }
